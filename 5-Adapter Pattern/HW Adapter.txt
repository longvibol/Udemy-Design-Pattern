1- Interface : PaymentProcessor

package com.vibol.corepattern.adapter05.hw;

public interface PaymentProcessor {

	Double pay(Double dollars);
}


2- PaymentProcessorImpl

package com.vibol.corepattern.adapter05.hw;

public class PaymentProcessorImpl implements PaymentProcessor{

	@Override
	public Double pay(Double dollars) {
		return dollars / 4000;
	}

}


3- PaymentAdapter


package com.vibol.corepattern.adapter05.hw;

public class PaymentAdapter {
	
	PaymentProcessor payment = new PaymentProcessorImpl();
	
	Double payInreal(Double i) {
		return payment.pay(i);		
	}
	

}

4- PaymentApp


package com.vibol.corepattern.adapter05.hw;

public class PaymentApp {

	public static void main(String[] args) {
		PaymentAdapter paymentAdapter = new PaymentAdapter();
		
		Double inDollars = 5000d;
		Double payInreal = paymentAdapter.payInreal(inDollars);
		
		System.out.println("Your input dollar is : "+ inDollars + " Equal In Real = " + payInreal + "r");

	}

}
